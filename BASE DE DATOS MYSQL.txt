CREATE DATABASE Calculadora;
USE Calculadora;
CREATE TABLE USUARIOS (
    CVEIDEN VARCHAR(50) PRIMARY KEY,
    PASSWORD VARCHAR(50)
);
CREATE TABLE EMPRESA (
    CVEIDEN VARCHAR(50) PRIMARY KEY,
    SECTOR VARCHAR(100),
    CAM VARCHAR(100),
    NombreEmpresa VARCHAR(200),
    SCIAN VARCHAR(100),
    UBICACION VARCHAR(300),
    CODIGOPOSTAL VARCHAR(10),
    MUNICIPIO VARCHAR(100),
    ENTIDADFEDERATIVA VARCHAR(100),
    LATITUD FLOAT,
    LONGITUD FLOAT,
    TELEFONO VARCHAR(20),
    FOREIGN KEY (CVEIDEN) REFERENCES USUARIOS(CVEIDEN)
);
CREATE TABLE MATERIAS_PRIMAS (
    CVEIDEN VARCHAR(50),
    MATERIAPRIMA VARCHAR(100),
    CONSUMO DECIMAL(18, 2),
    UNIDAD VARCHAR(50),
    FOREIGN KEY (CVEIDEN) REFERENCES Empresa(CVEIDEN)
);
CREATE TABLE PRODUCTOS (
    CVEIDEN VARCHAR(50),
    Producto VARCHAR(100),
    Cantidad DECIMAL(18, 2),
    Unidad VARCHAR(50) DEFAULT 'Tonelada',
    FOREIGN KEY (CVEIDEN) REFERENCES Empresa(CVEIDEN)
);
CREATE TABLE Combustibles (
    CVEIDEN VARCHAR(50),
    Combustible VARCHAR(100),
    Consumo DECIMAL(18, 2),
    Unidad VARCHAR(50) DEFAULT 'm^3',
    FOREIGN KEY (CVEIDEN) REFERENCES Empresa(CVEIDEN)
);
CREATE TABLE DatosInsumoEmpresarial (
    CVEIDEN VARCHAR(50),
    NEC INT,
    NAP INT,
    NCH INT,
    NECONT INT,
    FOREIGN KEY (CVEIDEN) REFERENCES Empresa(CVEIDEN)
);
CREATE TABLE ComponentesCombustibles (
    CVEIDEN VARCHAR(50),
    CO2_COMB_SC DECIMAL(18, 2),
    CH4_COMB_SC DECIMAL(18, 2),
    N2O_COMB_SC DECIMAL(18, 2),
    FOREIGN KEY (CVEIDEN) REFERENCES Empresa(CVEIDEN)
);
CREATE TABLE ComponentesParticulasProcesos (
    CVEIDEN VARCHAR(50),
    CO2_PROC_SC DECIMAL(18, 2),
    CH4_PROC_SC DECIMAL(18, 2),
    N2O_PROC_SC DECIMAL(18, 2),
    FOREIGN KEY (CVEIDEN) REFERENCES Empresa(CVEIDEN)
);
CREATE TABLE SumaParticulas (
    CVEIDEN VARCHAR(50),
    CO2_SUMA_SC DECIMAL(18, 2),
    CH4_SUMA_SC DECIMAL(18, 2),
    N2O_SUMA_SC DECIMAL(18, 2),
    FOREIGN KEY (CVEIDEN) REFERENCES Empresa(CVEIDEN)
);
CREATE TABLE DatosResponsableEmpresa (
    CVEIDEN VARCHAR(50),
    RESPONSABLE_TECNICO VARCHAR(200),
    ELABORADO VARCHAR(200),
    OBSERVACIONES VARCHAR(500),
    ZONA VARCHAR(100),
    FOREIGN KEY (CVEIDEN) REFERENCES Empresa(CVEIDEN)
);
CREATE TABLE Chimenea (
    CVEIDEN VARCHAR(50),
    CVECH VARCHAR(50),
    ALTURA DECIMAL(18, 2),
    DIAMETRO DECIMAL(18, 2),
    VELOCIDAD DECIMAL(18, 2),
    TEMPERATURA DECIMAL(18, 2),
    PRIMARY KEY (CVEIDEN, CVECH),
    FOREIGN KEY (CVEIDEN) REFERENCES Empresa(CVEIDEN)
);
CREATE TABLE CompuestosChimenea (
    CVEIDEN VARCHAR(50),
    CVECH VARCHAR(50),
    CO2 DECIMAL(18, 2),
    TIP_CO2 VARCHAR(100),
    CH4 DECIMAL(18, 2),
    TIP_CH4 VARCHAR(100),
    N2O DECIMAL(18, 2),
    TIP_N2O VARCHAR(100),
    FOREIGN KEY (CVEIDEN, CVECH) REFERENCES Chimenea(CVEIDEN, CVECH)
);
CREATE TABLE RegistroActividadesProceso (
    CVEIDEN VARCHAR(50),
    CVEACTP VARCHAR(50),
    TIPACTIV_GRAL VARCHAR(100),
    PROCESO VARCHAR(200),
    UNIDAD_FE VARCHAR(50),
    HR_ANO DECIMAL(18, 2),
    TIPO_DE_EMISION VARCHAR(100),
    EQUIPO_DE_CONTROL VARCHAR(100),
    CHIMENEA VARCHAR(100),
    FOREIGN KEY (CVEIDEN) REFERENCES Empresa(CVEIDEN)
);
CREATE TABLE ActividadParticulasProcesos (
    CVEIDEN VARCHAR(50),
    CO2_PROC_SC DECIMAL(18, 2),
    TIP_CO2 VARCHAR(100),
    CH4_PROC_SC DECIMAL(18, 2),
    TIP_CH4 VARCHAR(100),
    N2O_PROC_SC DECIMAL(18, 2),
    TIP_N2O VARCHAR(100),
    FOREIGN KEY (CVEIDEN) REFERENCES Empresa(CVEIDEN)
);
CREATE TABLE EquivalenteCombustible (
    CVEIDEN VARCHAR(50),
    CVEQPOC VARCHAR(50),
    TIPEQUIA VARCHAR(100),
    HR_ANO DECIMAL(18, 2),
    TIPO_DE_EMISION VARCHAR(100),
    CAPACIDAD DECIMAL(18, 2),
    UNIDAD_CAPACIDAD VARCHAR(50),
    TIPOCOMBUST VARCHAR(100),
    CANTIDAD DECIMAL(18, 2),
    UNIDAD_CANTIDAD VARCHAR(50),
    CHIMENEA VARCHAR(100),
    EQUIPO_DE_CONTROL VARCHAR(100),
    PORC_S DECIMAL(18, 2),
    FOREIGN KEY (CVEIDEN) REFERENCES Empresa(CVEIDEN)
);
CREATE TABLE EquivalenciaParticulasCombustible (
    CVEIDEN VARCHAR(50),
    CO2_COMB_SC DECIMAL(18, 2),
    TIP_CO2 VARCHAR(100),
    CH4_COMB_SC DECIMAL(18, 2),
    TIP_CH4 VARCHAR(100),
    N2O_COMB_SC DECIMAL(18, 2),
    TIP_N2O VARCHAR(100),
    FOREIGN KEY (CVEIDEN) REFERENCES Empresa(CVEIDEN)
);
CREATE TABLE EquivalenciasContaminantesControl (
    CVEIDEN VARCHAR(50),
    CVEQPOCON VARCHAR(50),
    TIPO_DE_EQ_CONTROL VARCHAR(100),
    EFICIENCIA DECIMAL(18, 2),
    METODO_DE_ESTIMACION VARCHAR(100),
    FOREIGN KEY (CVEIDEN) REFERENCES Empresa(CVEIDEN)
);
CREATE TABLE ControlContaminantes (
    CVEIDEN VARCHAR(50),
    CONTAM_CONTROL_1 VARCHAR(100),
    CONTAM_CONTROL_2 VARCHAR(100),
    CONTAM_CONTROL_3 VARCHAR(100),
    FOREIGN KEY (CVEIDEN) REFERENCES Empresa(CVEIDEN)
);
CREATE TABLE ActividadSuministro (
    CVEIDEN VARCHAR(50),
    ACTIVIDAD VARCHAR(100),
    EQUIPO_DE_COMBUSTION VARCHAR(100),
    SITUACION VARCHAR(100),
    MD VARCHAR(50),
    CT VARCHAR(50),
    CI VARCHAR(50),
    FOREIGN KEY (CVEIDEN) REFERENCES Empresa(CVEIDEN)
);
CREATE TABLE CombustiblesPoderesNetos (
    CVEIDEN VARCHAR(50),
    CVECH VARCHAR(50),
    TIPO_DE_COMBUSTIBLE VARCHAR(100),
    HR_ANO DECIMAL(18, 2),
    TIPO_DE_EMISION VARCHAR(100),
    CAPACIDAD DECIMAL(18, 2),
    UNIDAD VARCHAR(50),
    FOREIGN KEY (CVEIDEN, CVECH) REFERENCES Chimenea(CVEIDEN, CVECH)
);
CREATE TABLE TipoCombustiblePoderesNetos (
    CVEIDEN VARCHAR(50),
    TIPO_DE_COMBUSTIBLE VARCHAR(100),
    PODER_CALORIFICO_NETO DECIMAL(18, 2),
    UNIDAD_DE_MEDIDA_KJ_M3 DECIMAL(18, 2),
    UNIDAD_DE_MEDIDA_MJ_T DECIMAL(18, 2),
    CANTIDAD_DE_CONSUMO DECIMAL(18, 2),
    FACTOR_DE_CONVERSION_A_BEP DECIMAL(18, 2),
    EQUIVALENCIA_DE_UNIDADES_BEP_M3 DECIMAL(18, 2),
    EQUIVALENCIA_DE_UNIDADES_BEP_T DECIMAL(18, 2),
    CO2_T_MJ DECIMAL(18, 2),
    CH4_KG_MJ DECIMAL(18, 2),
    N2O_KG_MJ DECIMAL(18, 2),
    FOREIGN KEY (CVEIDEN) REFERENCES Empresa(CVEIDEN)
);
CREATE TABLE ProduccionUsoCFCs (
    CVEIDEN VARCHAR(50),
    CVECH VARCHAR(50),
    EQUIVALENTES_DE_ELECTRICIDAD DECIMAL(18, 2),
    HR_ANO DECIMAL(18, 2),
    TIPO_DE_EMISION VARCHAR(100),
    CAPACIDAD DECIMAL(18, 2),
    UNIDAD VARCHAR(50),
    FOREIGN KEY (CVEIDEN, CVECH) REFERENCES Chimenea(CVEIDEN, CVECH)
);
CREATE TABLE ProduccionCFCsHFCs (
    CVEIDEN VARCHAR(50),
    ACTIVIDAD VARCHAR(200),
    NOMBRE_DE_SUSTANCIA VARCHAR(100),
    MASA_SUSTANCIA_CONSUMIDA DECIMAL(18, 2),
    UNIDAD_MEDIDA VARCHAR(50),
    MASA_SUSTANCIA_ADICIONADA DECIMAL(18, 2),
    CANTIDAD DECIMAL(18, 2),
    UNIDAD VARCHAR(50),
    CO2 DECIMAL(18, 2),
    CH4 DECIMAL(18, 2),
    N2O DECIMAL(18, 2),
    FOREIGN KEY (CVEIDEN) REFERENCES Empresa(CVEIDEN)
);
CREATE TABLE EnergiaElectrica (
    CVEIDEN VARCHAR(50),
    CVECH VARCHAR(50),
    HR_ANO DECIMAL(18, 2),
    TIPO_DE_EMISION VARCHAR(100),
    CAPACIDAD DECIMAL(18, 2),
    UNIDAD VARCHAR(50),
    FOREIGN KEY (CVEIDEN, CVECH) REFERENCES Chimenea(CVEIDEN, CVECH)
);
CREATE TABLE ProduccionEnergiaElectrica (
    CVEIDEN VARCHAR(50),
    CAPACIDAD_INSTALADA DECIMAL(18, 2),
    TIPO_DE_PLANTA VARCHAR(100),
    GENERACION_ANUAL_MWH DECIMAL(18, 2),
    CONSUMO_COMBUSTIBLE VARCHAR(100),
    TIPO VARCHAR(100),
    CANTIDAD DECIMAL(18, 2),
    UNIDAD VARCHAR(50),
    CO2 DECIMAL(18, 2),
    CH4 DECIMAL(18, 2),
    N2O DECIMAL(18, 2),
    FOREIGN KEY (CVEIDEN) REFERENCES Empresa(CVEIDEN)
);
